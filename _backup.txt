function getMoonPhase(date) {
    const lp = 2551443; 
    const now = new Date(date.getFullYear(), date.getMonth(), date.getDate(), 20, 35, 0);
    const new_moon = new Date(1970, 0, 7, 20, 35, 0);
    const phase = ((now.getTime() - new_moon.getTime()) / 1000) % lp;
    const res = Math.floor(phase / (24 * 3600)) + 1;
    
    if (res <= 1 || res >= 29) return { name: "Nova", icon: "ðŸŒ‘" };
    if (res < 7) return { name: "Crescente", icon: "ðŸŒ’" };
    if (res < 14) return { name: "Crescente", icon: "ðŸŒ“" };
    if (res < 15) return { name: "Cheia", icon: "ðŸŒ•" };
    if (res < 22) return { name: "Minguante", icon: "ðŸŒ–" };
    return { name: "Minguante", icon: "ðŸŒ˜" };
}

function generateLuckyNumbers(seed) {
    // Simple hash deterministic function
    let hash = 0;
    for (let i = 0; i < seed.length; i++) {
        hash = ((hash << 5) - hash) + seed.charCodeAt(i);
        hash |= 0;
    }

    const numbers = [];
    let currentSeed = Math.abs(hash);
    
    while (numbers.length < 6) {
        currentSeed = (currentSeed * 16807) % 2147483647;
        let num = (currentSeed % 60) + 1;
        if (!numbers.includes(num)) {
            numbers.push(num);
        }
    }
    return numbers.sort((a, b) => a - b);
}
